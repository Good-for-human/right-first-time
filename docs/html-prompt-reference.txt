    blue: "bg-blue-50 text-blue-700 border-blue-200",

  return (

const SidebarGroup = ({ title, queue, icon: Icon, view, activeTaskId, setActiveTaskId, setView }) => {
  if (!queue || queue.length === 0) return null;
  
  return (
    <div className="mb-5">
      <div className="text-[11px] font-bold text-slate-400 uppercase tracking-wider mb-2 px-4 flex items-center gap-1.5">
        {Icon && <Icon size={12} />} {title}
      </div>
      <div className="space-y-0.5 px-2">
        {queue.map(task => (
          <button
            key={task.id}
            onClick={() => { setActiveTaskId(task.id); setView('workspace'); }}
            className={`w-full text-left px-3 py-2.5 rounded-lg text-sm transition flex items-start justify-between group ${
              view === 'workspace' && activeTaskId === task.id 
                ? 'bg-blue-50/70 border border-blue-100 shadow-sm' 
                : 'hover:bg-slate-50 border border-transparent'
            }`}
          >
            <div className="flex items-start gap-2.5 truncate w-full">
              <div className="mt-1 shrink-0">{renderStatusIcon(task.status)}</div>
              <div className="truncate flex-1 min-w-0">
                <div className={`font-medium truncate ${view === 'workspace' && activeTaskId === task.id ? 'text-[#0052D9]' : 'text-slate-700'}`}>
                  {task.name || task.asin}
                </div>
                {task.name && (
                  <div className={`text-xs truncate mt-0.5 ${view === 'workspace' && activeTaskId === task.id ? 'text-blue-500/80' : 'text-slate-400'}`}>
                    {task.asin}
                  </div>
                )}
              </div>
            </div>
          </button>
        ))}
      </div>
    </div>
  );
};

  const [newCategory, setNewCategory] = useState('');
  
  const [isSavingGlobal, setIsSavingGlobal] = useState(false);
  const [globalSaved, setGlobalSaved] = useState(false);

  const [ruleModal, setRuleModal] = useState({ open: false, type: 'instruction', data: null });
  const [formRule, setFormRule] = useState({ name: '', severity: 'High', priority: 'Required', targetSection: 'all', referenceAsins: [] });
  const [isOptimizingRule, setIsOptimizingRule] = useState(false);
  
  const [personaModal, setPersonaModal] = useState({ open: false, data: null });
  const [formPersona, setFormPersona] = useState({ name: '', description: '' });
  const [isOptimizingPersona, setIsOptimizingPersona] = useState(false);

  const [deleteConfirm, setDeleteConfirm] = useState({ open: false, type: '', payload: null, name: '' });

  const filteredRules = rules.filter(r => r.category === activeCategory);
  const archivedTasks = tasks.filter(tObj => tObj.status === 'archived');

  const getSectionLabel = (section) => {
    switch(section) {
      case 'title': return t('section.title');
      case 'bullets': return t('section.bullets');
      case 'description': return t('section.desc');
      default: return t('modal.scopeAll');
    }
  };

  const handleAddCategory = () => {
    if (newCategory.trim() && !categories.includes(newCategory.trim())) {
      setCategories([...categories, newCategory.trim()]);
      setActiveCategory(newCategory.trim());
      setNewCategory('');
      setIsAddingCategory(false);
    }
  };

  const executeDelete = () => {
    if (deleteConfirm.type === 'category') {
      const catToDelete = deleteConfirm.payload;
      setCategories(categories.filter(c => c !== catToDelete));
      setRules(rules.filter(r => r.category !== catToDelete));
      if (activeCategory === catToDelete) setActiveCategory('通用');
    } else if (deleteConfirm.type === 'persona') {
      setPersonas(personas.filter(p => p.id !== deleteConfirm.payload));
    } else if (deleteConfirm.type === 'rule') {
      setRules(rules.filter(r => r.id !== deleteConfirm.payload));
    }
    setDeleteConfirm({ open: false, type: '', payload: null, name: '' });
  };

  const handleGlobalSave = () => {
    setIsSavingGlobal(true);
    setTimeout(() => {
      setIsSavingGlobal(false);
      setGlobalSaved(true);
      setAppSettings({ 
        ...appSettings, 
        isSaved: !!appSettings.apiKey,
        isTinyfishSaved: !!appSettings.tinyfishApiKey

    }, 600);
  };

  const openRuleModal = (type, existingRule = null) => {
    setFormRule(existingRule ? { ...existingRule, referenceAsins: existingRule.referenceAsins || [] } : { name: '', severity: 'High', priority: 'Required', targetSection: 'all', referenceAsins: [] });
    setRuleModal({ open: true, type, data: existingRule });
  };

  const saveRule = () => {
    if (!formRule.name.trim()) return;
    if (ruleModal.data) {
      setRules(rules.map(r => r.id === ruleModal.data.id ? { ...r, ...formRule } : r));
    } else {
      setRules([...rules, { 
        id: Date.now(), category: activeCategory, type: ruleModal.type,

  const [activeTaskId, setActiveTaskId] = useState(INITIAL_TASKS[0].id);
  const [view, setView] = useState('workspace'); 
  const [searchTerm, setSearchTerm] = useState('');

  const [isModalOpen, setIsModalOpen] = useState(false);
  const [newTaskAsin, setNewTaskAsin] = useState('');
  const [newTaskUrl, setNewTaskUrl] = useState('');
  const [newTaskCategory, setNewTaskCategory] = useState('通用');

  const activeTask = tasks.find(tObj => tObj.id === activeTaskId);

  const updateTask = useCallback((id, updates) => {
    setTasks(prevTasks => prevTasks.map(tObj => tObj.id === id ? { ...tObj, ...updates } : tObj));
  }, []);

  const handleCreateTask = (e) => {
    e.preventDefault();
    if (!newTaskAsin.trim()) return;

    const newTask = {
      id: Date.now().toString(),
      asin: newTaskAsin.toUpperCase(),
      name: '', 
      category: newTaskCategory,
      language: appSettings.targetLanguage, 
      personaIds: [],
      url: newTaskUrl.trim(),
      status: 'pending' 
    };
    setTasks([newTask, ...tasks]);
    setActiveTaskId(newTask.id);
    setView('workspace');
    setIsModalOpen(false);
    setNewTaskAsin('');
    setNewTaskUrl('');
    setNewTaskCategory('通用');
  };

  const filterBySearch = (tObj) => tObj.asin.toLowerCase().includes(searchTerm.toLowerCase()) || (tObj.name && tObj.name.toLowerCase().includes(searchTerm.toLowerCase()));
  const activeQueue = tasks.filter(tObj => ['pending', 'fetched', 'review'].includes(tObj.status) && filterBySearch(tObj));
  const archivedQueue = tasks.filter(tObj => tObj.status === 'archived' && filterBySearch(tObj));

  return (
    <div className="h-screen flex bg-[#F7F9FC] font-sans text-slate-900 overflow-hidden">
      
      <div className="w-64 bg-white border-r border-slate-200 flex flex-col shrink-0 z-20 shadow-sm">
        <div className="p-5 border-b border-slate-100 flex items-center gap-2">
          <Zap size={22} className="text-[#0052D9] fill-current" />
          <h1 className="text-lg font-bold tracking-tight text-[#0052D9]">{t('app.title')}</h1>
        </div>

        <div className="p-4 border-b border-slate-100 bg-slate-50/50">
          <button 
            onClick={() => setIsModalOpen(true)}
            className="w-full bg-[#0052D9] text-white py-2 rounded-md text-[13px] font-semibold hover:bg-blue-800 transition flex justify-center items-center gap-2 shadow-sm"
          >
            <Plus size={16} /> {t('sidebar.newTask')}
          </button>
        </div>

        <div className="flex-1 overflow-y-auto flex flex-col">
          <div className="p-3 border-b border-slate-100 shrink-0 sticky top-0 bg-white z-10">
            <div className="relative">
              <Search size={14} className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" />
              <input 
                type="text" 
                placeholder={t('sidebar.search')} 
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="w-full pl-8 pr-3 py-2 bg-slate-50 border border-slate-200 rounded-md text-xs focus:ring-1 focus:ring-[#0052D9] outline-none transition-all shadow-inner"
              />
            </div>
          </div>

          <div className="py-4 flex-1 overflow-y-auto">
            <SidebarGroup title={t('sidebar.review')} queue={activeQueue} icon={LayoutDashboard} view={view} activeTaskId={activeTaskId} setActiveTaskId={setActiveTaskId} setView={setView} />
            <SidebarGroup title={t('sidebar.archived')} queue={archivedQueue} icon={Archive} view={view} activeTaskId={activeTaskId} setActiveTaskId={setActiveTaskId} setView={setView} />
            {tasks.filter(filterBySearch).length === 0 && <div className="text-xs text-center text-slate-400 py-6">{t('sidebar.noResults')}</div>}
          </div>
        </div>

        <div className="p-3 border-t border-slate-100 bg-slate-50 shrink-0">
          <button
            onClick={() => setView('rules')}
            className={`w-full text-left px-3 py-2.5 rounded-lg text-sm font-medium transition flex items-center gap-2.5 ${
              view === 'rules' ? 'bg-blue-100 text-[#0052D9]' : 'text-slate-600 hover:bg-white hover:shadow-sm border border-transparent'
            }`}
          >
            <Settings size={16} />
            {t('sidebar.settings')}
          </button>
        </div>
      </div>

      <div className="flex-1 flex flex-col min-w-0">
        {view === 'workspace' ? (
          <Workspace 
            task={activeTask} 
            updateTask={updateTask} 
            categories={categories}
            personas={personas}
            appSettings={appSettings} 
            setAppSettings={setAppSettings} 
            t={t}
          />
        ) : (
          <SettingsAndRules 
            categories={categories} 
            setCategories={setCategories}
            rules={rules}
            setRules={setRules}
            personas={personas}
            setPersonas={setPersonas}
            appSettings={appSettings}
            setAppSettings={setAppSettings}
            tasks={tasks}
            t={t}
          />
        )}
      </div>

      {isModalOpen && (
        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm z-50 flex items-center justify-center animate-in fade-in overflow-y-auto py-10">
          <div className="bg-white rounded-2xl shadow-xl w-full max-w-sm overflow-hidden animate-in zoom-in-95 my-auto">
            <div className="px-6 py-4 border-b border-slate-100 flex justify-between items-center bg-slate-50/50">
              <h3 className="font-bold text-slate-800">{t('modal.createTask')}</h3>
              <button onClick={() => setIsModalOpen(false)} className="text-slate-400 hover:text-slate-600 text-2xl leading-none font-light transition">
                <X size={20} />
              </button>
            </div>
            <form onSubmit={handleCreateTask} className="p-6 space-y-4">
              <div>
                <label className="block text-[13px] font-medium text-slate-700 mb-1.5">{t('modal.asin')}</label>
                <input 
                  type="text" 
                  autoFocus
                  required
                  placeholder="e.g. B09X1Y2Z3W"
                  value={newTaskAsin}
                  onChange={e => setNewTaskAsin(e.target.value)}
                  className="w-full border border-slate-300 rounded-lg px-3 py-2 text-sm focus:border-[#0052D9] outline-none shadow-inner"
                />
              </div>
              <div>
                <label className="block text-[13px] font-medium text-slate-700 mb-1.5">{t('modal.url')}</label>
                <input 
                  type="url" 
                  placeholder={t('modal.urlPlaceholder')}
                  value={newTaskUrl}
                  onChange={e => setNewTaskUrl(e.target.value)}
                  className="w-full border border-slate-300 rounded-lg px-3 py-2 text-sm focus:border-[#0052D9] outline-none shadow-inner"
                />
              </div>
              <div>
                <label className="block text-[13px] font-medium text-slate-700 mb-1.5">{t('modal.category')}</label>
                <select 
                  value={newTaskCategory}
                  onChange={e => setNewTaskCategory(e.target.value)}
                  className="w-full border border-slate-300 rounded-lg px-3 py-2 text-sm focus:border-[#0052D9] outline-none bg-white shadow-inner"
                >
                  {categories.map(cat => (
                    <option key={cat} value={cat}>{cat}</option>
                  ))}
                </select>
              </div>
              <div className="pt-4 flex justify-end gap-3 border-t border-slate-100">
                <button 
                  type="button" 
                  onClick={() => setIsModalOpen(false)}
                  className="px-4 py-2 text-sm font-medium text-slate-600 hover:bg-slate-100 rounded-md transition"
                >
                  {t('modal.cancel')}
                </button>
                <button 
                  type="submit"
                  className="px-5 py-2 text-sm font-medium text-white bg-[#0052D9] hover:bg-blue-800 rounded-md transition shadow-sm"
                >
                  {t('modal.create')}
                </button>
              </div>
            </form>
          </div>
        </div>
      )}

    </div>
  );
}
